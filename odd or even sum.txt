import java.io.*;
import java.util.*;
class UserMainCode
{

public int EvenOddDigitSum(int input1,String input2){
int digit;
int even=0;
int odd=0;
while(input1!=0){
digit=input1%10;
if(digit%2==0)
even+=digit;
else
odd+=digit;
input1=input1/10;
}
if(input2.equals("even"))
return(even);
else
return(odd);
}
}
